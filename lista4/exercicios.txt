import hashlib
from cryptography.fernet import Fernet

def gerear_hash_senha(senha):
    hash_objeto  = hashlib.sha256()
    hash_objeto.update(senha.encode())
    hash_objeto.hexdigest()
senha = input("Digite uma senha ")
hash_senha = gerear_hash_senha(senha)
print(f"Hash SHA-256 de senha :{hash_senha}")
